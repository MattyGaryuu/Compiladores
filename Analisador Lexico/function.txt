program Exemplo;

var
  numeros: array[1..5] of integer;

function SomaArray(arr: array of integer; size: integer): integer;
var
  i, soma: integer;
begin
  soma := 0;
  for i := 0 to size - 1 do
  begin
    soma := soma + arr[i];
  end;
  SomaArray := soma;
end;

procedure PreencherArray(var arr: array of integer; size: integer);
var
  i: integer;
begin
  for i := 0 to size - 1 do
  begin
    write('Digite um número: ');
    read(arr[i]);
  end;
end;

procedure ImprimirArray(arr: array of integer; size: integer);
var
  i: integer;
begin
  writeln('Conteúdo do array:');
  for i := 0 to size - 1 do
  begin
    write(arr[i]);
  end;
end;

begin
  PreencherArray(numeros, 5);
  ImprimirArray(numeros, 5);
  write('A soma dos números do array é: ', SomaArray(numeros, 5));
  write('Fim do programa.');
end.
